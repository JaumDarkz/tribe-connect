        twitterUser.username
      );

        user_id: user.id,
        tweet_id: tweet.id,
        engagement_type: type
      });

      if (existing) continue;

      // Award points
      await this.awardPoints(user.id, tweet.id, type, points);
    }
  }

    await this.db.engagements.create({

      engagement_type: engagementType,

    await this.db.pointsTransactions.create({
      user_id: userId,
      amount: points,
      transaction_type: 'earned',
      source: `tweet_${engagementType}`
    });

    // Send Discord notification (optional)
    await this.notifyUser(userId, points, engagementType);
  }

class VoiceTracker {
  constructor() {
    this.activeVoiceSessions = new Map(); // userId -> session data
    this.pointInterval = 60000; // Award every minute
  }

  // Track when user joins voice
  async handleVoiceJoin(userId, channelId) {
    this.activeVoiceSessions.set(userId, {
      channelId,
      joinedAt: Date.now(),
      lastPointAward: Date.now(),
      isDeafened: false,
      isStreaming: false
    });

    // Start point accumulation
    this.startPointAccumulation(userId);
  }

  // Track voice state updates (mute, deafen, streaming)
  async handleVoiceStateUpdate(userId, oldState, newState) {
    const session = this.activeVoiceSessions.get(userId);
    if (!session) return;

    // Update session state
    session.isDeafened = newState.selfDeaf || newState.serverDeaf;
    session.isStreaming = newState.streaming;

    // If user became deafened, stop awarding points
    if (session.isDeafened) {
      clearInterval(session.intervalId);
    } else if (!session.intervalId) {
      // Restarted if they undeafen
      this.startPointAccumulation(userId);
    }
  }

  // Award points every minute
  startPointAccumulation(userId) {
    const session = this.activeVoiceSessions.get(userId);

    session.intervalId = setInterval(async () => {
      const currentSession = this.activeVoiceSessions.get(userId);

      // Don't award if deafened (AFK detection)
      if (currentSession.isDeafened) return;

      // Base points
      let points = 5;

      // Streaming bonus
      if (currentSession.isStreaming) {
        points += 2;
      }

      // Award points
      await this.awardVoicePoints(userId, points);
      currentSession.lastPointAward = Date.now();

    }, this.pointInterval);
  }

  // Handle voice leave
  async handleVoiceLeave(userId) {
    const session = this.activeVoiceSessions.get(userId);
    if (!session) return;

    // Clear interval
    if (session.intervalId) {
      clearInterval(session.intervalId);
    }

    // Calculate total session time
    const sessionDuration = Date.now() - session.joinedAt;
    const minutes = Math.floor(sessionDuration / 60000);

    // Log session
    await db.voiceSessions.create({
      user_id: userId,
      channel_id: session.channelId,
      duration_minutes: minutes,
      ended_at: new Date()
    });

    // Remove from active sessions
    this.activeVoiceSessions.delete(userId);
  }

  async awardVoicePoints(userId, points) {
    // Apply role multipliers
    const finalPoints = await calculatePointsWithMultiplier(userId, points);

    // Award
    await awardPoints(userId, finalPoints, 'voice_activity');
  }
}

      created_at: { $lt: new Date(Date.now() - 300000) } // 5 min delay
    });

    for (const task of pending) {
      try {
        let verified = false;

        switch (task.platform) {
          case 'twitter':
            verified = await this.verifyTwitterAction(task);
            break;
          case 'youtube':
            verified = await this.verifyYouTubeAction(task);
            break;
          case 'tiktok':
            verified = await this.verifyTikTokAction(task);
            break;
        }

        if (verified) {
          // Award points
          const points = this.getPointsForAction(task.action_type);
          await awardPoints(task.user_id, points, `${task.platform}_${task.action_type}`);

          // Update status
          await db.verificationQueue.update(task.id, {
            status: 'verified',
            verified_at: new Date()
          });
        } else {
          // Retry later or mark as failed after X attempts
          await db.verificationQueue.increment('retry_count', 1, { id: task.id });
        }
      } catch (error) {
        console.error(`Verification failed for task ${task.id}:`, error);
      }
    }
  }
}

  async verifyWalletOwnership(userId, walletAddress, signature, message) {
    try {
      // For Ethereum
      if (walletAddress.startsWith('0x')) {
        const recoveredAddress = ethers.utils.verifyMessage(message, signature);
        return recoveredAddress.toLowerCase() === walletAddress.toLowerCase();
      }

      // For Solana
      else {
        const messageBytes = new TextEncoder().encode(message);
        const signatureBytes = Buffer.from(signature, 'base64');
        const publicKey = new PublicKey(walletAddress);

        return await nacl.sign.detached.verify(
          messageBytes,
          signatureBytes,
          publicKey.toBytes()
        );
      }
    } catch (error) {
      console.error('Wallet verification failed:', error);
      return false;
    }
  }

  // Store verified wallet
  async linkWallet(userId, walletAddress, chain) {
    await db.wallets.create({
      user_id: userId,
      wallet_address: walletAddress,
      chain, // 'ethereum', 'solana', 'bitcoin'
      verified_at: new Date()
    });
  }

  // Token gating
  async checkTokenOwnership(walletAddress, tokenAddress, minimumAmount) {
    try {
      // For Solana SPL tokens
      if (tokenAddress.length === 44) {
        const publicKey = new PublicKey(walletAddress);
        const tokenAccounts = await this.solanaConnection.getParsedTokenAccountsByOwner(
          publicKey,
          { mint: new PublicKey(tokenAddress) }
        );

        if (tokenAccounts.value.length === 0) return false;

        const balance = tokenAccounts.value[0].account.data.parsed.info.tokenAmount.uiAmount;
        return balance >= minimumAmount;
      }

      // For Ethereum ERC-20
      else if (tokenAddress.startsWith('0x')) {
        const provider = new ethers.providers.JsonRpcProvider(process.env.ETH_RPC_URL);
        const contract = new ethers.Contract(
          tokenAddress,
          ['function balanceOf(address) view returns (uint256)'],
          provider
        );

        const balance = await contract.balanceOf(walletAddress);
        return balance.gte(ethers.utils.parseEther(minimumAmount.toString()));
      }
    } catch (error) {
      console.error('Token verification failed:', error);
      return false;
    }
  }

  // Grant role based on token ownership
  async grantTokenGatedRole(userId, roleId, tokenRequirements) {
    const wallets = await db.wallets.findAll({ user_id: userId });

    for (const wallet of wallets) {
      const ownsToken = await this.checkTokenOwnership(
        wallet.wallet_address,
        tokenRequirements.token_address,
        tokenRequirements.minimum_amount
      );

      if (ownsToken) {
        // Grant role
        const member = await discordClient.guilds.cache
          .first()
          .members.fetch(userId);

        await member.roles.add(roleId);
        return true;
      }
    }

    return false;
  }
}

class AutoFarmSystem {
  constructor() {

  }

  // Start auto-farming for premium user
  async startAutoFarm(userId, serverId) {
    // Verify premium status
    const user = await db.users.findById(userId);
    if (!user.is_premium) {
      throw new Error('Premium subscription required');
    }

    // Check farm limit based on tier
    const activeServers = await db.farmingSessions
      .where('user_id', userId)
      .where('status', 'active')
      .count();

    const maxServers = this.getMaxFarmingServers(user.premium_tier);
    if (activeServers >= maxServers) {
      throw new Error(`Maximum ${maxServers} servers for your tier`);
    }

    // Create farming session
    const session = await db.farmingSessions.create({
      user_id: userId,
      server_id: serverId,
      status: 'active',
      started_at: new Date()
    });

    // Start automated point collection
    this.runFarmingLoop(session.id, userId, serverId);

    return session;
  }

  async runFarmingLoop(sessionId, userId, serverId) {
    const intervalId = setInterval(async () => {
      try {
        const session = await db.farmingSessions.findById(sessionId);
        if (session.status !== 'active') {
          clearInterval(intervalId);
          return;
        }

        // Simulate various activities
        await this.simulateDailyClaim(userId);
        await this.simulateMessageEngagement(userId, serverId);
        await this.claimAutoRewards(userId, serverId);

      } catch (error) {
        console.error(`Auto-farm error for session ${sessionId}:`, error);
      }
    }, 3600000); // Every hour

    this.activeFarmingSessions.set(sessionId, intervalId);
  }

  async simulateDailyClaim(userId) {
    const user = await db.users.findById(userId);
    const lastClaim = user.last_daily_claim;

    // Check if can claim
    if (!lastClaim || !isSameDay(lastClaim, new Date())) {
      await processDailyClaimPoints(userId);
    }
  }

  async claimAutoRewards(userId, serverId) {
    // Check for any pending automated rewards
    const pendingRewards = await db.automatedRewards
      .where('user_id', userId)
      .where('server_id', serverId)
      .where('status', 'pending')
      .findAll();

    for (const reward of pendingRewards) {
      await awardPoints(userId, reward.points, 'auto_farm');
      await db.automatedRewards.update(reward.id, { status: 'claimed' });
    }
  }

  getMaxFarmingServers(premiumTier) {
    const limits = {
      'gold': 1,
      'diamond': 3
    };
    return limits[premiumTier] || 0;
  }
}
